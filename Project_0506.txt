PROGRAM FilmCollectionManagement

DEKLARASI:
    films_0508 ← List kosong untuk menampung Data Film
    genreFilm_0508_0508 ← ["Aksi", "Komedi", "Drama", "Horror", "Romantis", "Sci-Fi", "Dokumenter"]

FUNCTION Main():
    TRY:
        WHILE true:
            CLEAR screen
            TAMPILKAN "======================================="
            TAMPILKAN "==| APLIKASI MANAJEMEN KOLEKSI FILM |=="
            TAMPILKAN "======================================="
            TAMPILKAN "||   PILIH MENU                      ||"
            TAMPILKAN "||   1. Tambah Film                  ||"
            TAMPILKAN "||   2. Lihat Data Film              ||"
            TAMPILKAN "||   3. Update Data Film             ||"
            TAMPILKAN "||   4. Hapus Data Film              ||"
            TAMPILKAN "||   5. Cari Film (judul)            ||"
            TAMPILKAN "||   6. Filter berdasarkan Genre     ||"
            TAMPILKAN "||   0. Keluar                       ||"
            TAMPILKAN "======================================="
            BACA pilihan_0508 from user (convert to integer)
            
            IF pilihan_0508 = 1 THEN
                TambahFilm()
            ELSE IF pilihan_0508 = 2 THEN
                LihatDataFilm()
            ELSE IF pilihan_0508 = 3 THEN
                UpdateFilm()
            ELSE IF pilihan_0508 = 4 THEN
                HapusFilm()
            ELSE IF pilihan_0508 = 5 THEN
                CariFilm()
            ELSE IF pilihan_0508 = 6 THEN
                FilterGenre()
            ELSE IF pilihan_0508 = 0 THEN
                TAMPILKAN "Keluar... Terima kasih."
                BREAK from loop
            ELSE:
                TAMPILKAN "Pilihan_0508 tidak valid."
            
            TAMPILKAN "Tekan Enter untuk kembali ke menu..."
            BACA input
    CATCH error:
        TAMPILKAN "Terjadi kesalahan dalam aplikasi."

FUNCTION TambahFilm():
    CLEAR screen
    TAMPILKAN header "APLIKASI MANAJEMEN KOLEKSI FILM"
    TAMPILKAN "(TAMBAH ENTRI FILM BARU)"
    
    // INPUT JUDUL FILM
    REPEAT:
        BACA judulFilm_0508
        IF judulFilm_0508 is empty or whitespace only THEN
            TAMPILKAN "Judul film tidak boleh kosong."
    UNTIL judulFilm_0508 is valid
    
    // INPUT GENRE
    TAMPILKAN "=== Pilih Genre ==="
    FOR i ← 0 to genreFilm_0508_0508.count - 1:
        TAMPILKAN (i + 1) + ". " + genreFilm_0508_0508[i]
    
    REPEAT:
        BACA genreInput_0508
        TRY:
            nomor_0508Genre_0508 ← convert genreInput_0508 to integer
            IF nomor_0508Genre_0508 >= 1 AND nomor_0508Genre_0508 <= genreFilm_0508_0508.count THEN
                genreFilm_0508 ← genreFilm_0508_0508[nomor_0508Genre_0508 - 1]
                BREAK from loop
            ELSE:
                TAMPILKAN "Pilihan_0508 genre tidak valid. Silakan pilih nomor_0508 genre yang benar."
        CATCH FormatException:
            TAMPILKAN "Pilihan_0508 genre tidak valid. Silakan pilih nomor_0508 genre yang benar."
    UNTIL valid
    
    // INPUT TAHUN RILIS
    TAMPILKAN "=== Masukkan Tahun Rilis ==="
    REPEAT:
        BACA tahunInput_0508
        TRY:
            tahunRilis_0508 ← convert tahunInput_0508 to integer
            IF tahunRilis_0508 > 1800 AND tahunRilis_0508 <= current_year THEN
                BREAK from loop
            ELSE:
                TAMPILKAN "Tahun rilis tidak valid. Masukkan tahun yang benar (misalnya: 1999)."
        CATCH FormatException:
            TAMPILKAN "Tahun rilis tidak valid. Masukkan tahun yang benar (misalnya: 1999)."
    UNTIL valid
    
    // ADD FILM TO COLLECTION
    filmBaru_0508 ← "Judul: " + judulFilm_0508 + ", Genre: " + genreFilm_0508 + ", Tahun: " + tahunRilis_0508
    films_0508.ADD(filmBaru_0508)
    TAMPILKAN "Film '" + judulFilm_0508 + "' (" + genreFilm_0508 + ", " + tahunRilis_0508 + ") berhasil ditambahkan."

FUNCTION LihatDataFilm():
    CLEAR screen
    TAMPILKAN header "APLIKASI MANAJEMEN KOLEKSI FILM"
    TAMPILKAN "(DAFTAR FILM)"
    
    IF films_0508.count = 0 THEN
        TAMPILKAN "Belum ada film."
        RETURN
    
    nomor_0508 ← 1
    FOREACH film_0508 IN films_0508:
        TAMPILKAN nomor_0508 + ". " + film_0508
        nomor_0508 ← nomor_0508 + 1

FUNCTION UpdateFilm():
    CLEAR screen
    TAMPILKAN header "APLIKASI MANAJEMEN KOLEKSI FILM"
    TAMPILKAN "(UPDATE ENTRI FILM)"
    
    IF films_0508.count = 0 THEN
        TAMPILKAN "Belum ada film."
        RETURN
    
    LihatDataFilm()
    
    TAMPILKAN "Pilih nomor film yang akan diupdate: "
    BACA inputNomorFilm_0508
    nomorFilm_0508 ← -1
    
    REPEAT:
        TRY:
            nomorFilm_0508 ← convert inputNomorFilm_0508 to integer
            IF nomorFilm_0508 < 1 OR nomorFilm_0508 > films_0508.count THEN
                TAMPILKAN "Nomor film tidak valid. Pilih nomor yang sesuai."
                BACA inputNomorFilm_0508
            ELSE:
                BREAK from loop
        CATCH:
            TAMPILKAN "Input tidak valid. Masukkan angka yang benar."
            BACA inputNomorFilm_0508
    UNTIL valid
    
    filmYangDipilih_0508 ← films_0508[nomorFilm_0508 - 1]
    TAMPILKAN "Film yang dipilih: " + filmYangDipilih_0508
    
    // UPDATE JUDUL
    BACA judulBaru_0508 (with prompt "Masukkan Judul Baru (kosongkan untuk tidak mengubah): ")
    IF judulBaru_0508 not empty THEN
        bagianFilm ← filmYangDipilih_0508.SPLIT(",")
        bagianFilm[0] ← "Judul: " + judulBaru_0508
        filmYangDipilih_0508 ← bagianFilm.JOIN(", ")
    
    // UPDATE GENRE
    BACA genreBaru_0508 (with prompt "Masukkan Genre Baru (kosongkan untuk tidak mengubah): ")
    IF genreBaru_0508 not whitespace THEN
        bagianFilm ← filmYangDipilih_0508.SPLIT(",")
        bagianFilm[1] ← "Genre: " + genreBaru_0508
        filmYangDipilih_0508 ← bagianFilm.JOIN(", ")
    
    // UPDATE TAHUN RILIS
    BACA tahunRilisInput_0508 (with prompt "Masukkan Tahun Rilis Baru (kosongkan untuk tidak mengubah): ")
    IF tahunRilisInput_0508 not whitespace THEN
        TRY:
            tahunRilis_0508 ← convert tahunRilisInput_0508 to integer
            bagianFilm ← filmYangDipilih_0508.SPLIT(",")
            bagianFilm[2] ← "Tahun: " + tahunRilis_0508
            filmYangDipilih_0508 ← bagianFilm.JOIN(", ")
        CATCH:
            TAMPILKAN "Tahun rilis tidak valid. Masukkan tahun yang benar (misalnya: 1999)."
    
    // SAVE CHANGES
    films_0508[nomorFilm_0508 - 1] ← filmYangDipilih_0508
    TAMPILKAN "Data film berhasil diupdate."

FUNCTION HapusFilm():
    CLEAR screen
    TAMPILKAN header "APLIKASI MANAJEMEN KOLEKSI FILM"
    TAMPILKAN "(HAPUS FILM)"
    
    IF films_0508.count = 0 THEN
        TAMPILKAN "Belum ada film."
        RETURN
    
    LihatDataFilm()
    
    TAMPILKAN "Pilih nomor film yang akan dihapus: "
    BACA inputNomorFilm
    
    IF convert inputNomorFilm to integer FAILS OR inputNomorFilm < 1 OR inputNomorFilm > films_0508.count THEN
        TAMPILKAN "Pilihan_0508 tidak valid."
        RETURN
    
    nomorFilm ← convert inputNomorFilm to integer
    filmYangDihapus ← films_0508[nomorFilm - 1]
    films_0508.REMOVE_AT(nomorFilm - 1)
    
    TAMPILKAN "Film '" + filmYangDihapus + "' berhasil dihapus."

FUNCTION CariFilm():
    BACA kataKunci (with prompt "Masukkan kata kunci judul untuk mencari: ")
    kataKunci ← kataKunci.TRIM()
    
    hasilPencarian ← films_0508.FIND_ALL(film contains kataKunci, case-insensitive)
    
    TAMPILKAN "Hasil pencarian (" + hasilPencarian.count + "):"
    
    IF hasilPencarian.count = 0 THEN
        TAMPILKAN "Tidak ditemukan film yang sesuai dengan kata kunci."
        RETURN
    
    FOR urutanFilm ← 0 to hasilPencarian.count - 1:
        TAMPILKAN (urutanFilm + 1) + ". " + hasilPencarian[urutanFilm]

FUNCTION FilterGenre():
    TAMPILKAN "== Filter berdasarkan Genre =="
    TAMPILKAN "Daftar genre:"
    
    FOR i ← 0 to genreFilm_0508_0508.count - 1:
        TAMPILKAN (i + 1) + ". " + genreFilm_0508_0508[i]
    
    BACA inputNomorGenre (with prompt "Pilih nomor genre: ")
    nomorGenre ← -1
    
    REPEAT:
        TRY:
            nomorGenre ← convert inputNomorGenre to integer
            IF nomorGenre < 1 OR nomorGenre > genreFilm_0508_0508.count THEN
                TAMPILKAN "Nomor genre tidak valid. Pilih nomor genre yang sesuai."
                BACA inputNomorGenre
            ELSE:
                BREAK from loop
        CATCH FormatException:
            TAMPILKAN "Input tidak valid. Masukkan angka yang benar."
            BACA inputNomorGenre
    UNTIL valid
    
    genre ← genreFilm_0508_0508[nomorGenre - 1]
    
    hasilPencarian ← films_0508.FIND_ALL(film contains genre, case-insensitive)
    
    TAMPILKAN "Film dengan genre '" + genre + "' (" + hasilPencarian.count + "):"
    
    IF hasilPencarian.count = 0 THEN
        TAMPILKAN "Tidak ada film."
    ELSE:
        FOR i ← 0 to hasilPencarian.count - 1:
            TAMPILKAN (i + 1) + ". " + hasilPencarian[i]
